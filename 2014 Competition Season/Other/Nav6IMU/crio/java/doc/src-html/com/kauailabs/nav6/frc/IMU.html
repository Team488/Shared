022/**
023 * The IMU class provides a simplified interface to the KauaiLabs nav6 IMU.
024 * 
025 * The IMU class enables access to basic connectivity and state information, 
026 * as well as key orientation information (yaw, pitch, roll, compass heading).
027 * 
028 * Advanced capabilities of the nav6 IMU may be accessed via the IMUAdvanced 
029 * class.
030 * @author Scott
031 */
032public class IMU extends SensorBase implements PIDSource, LiveWindowSendable, Runnable {
033
034    static final int    YAW_HISTORY_LENGTH      = 10;
035    static final byte   DEFAULT_UPDATE_RATE_HZ  = 100;
036    static final short  DEFAULT_ACCEL_FSR_G     = 2;
037    static final short  DEFAULT_GYRO_FSR_DPS    = 2000;
038    
039    BufferingSerialPort serial_port;
040    float yaw_history[];
041    int next_yaw_history_index;
042    double user_yaw_offset;
043    ITable m_table;
044    Thread m_thread;
045    protected byte update_rate_hz;

048    volatile float pitch;
049    volatile float roll;

051    volatile int update_count = 0;
052    volatile int byte_count = 0;
053    volatile float nav6_yaw_offset_degrees;
054    volatile short accel_fsr_g;
055    volatile short gyro_fsr_dps;
056    volatile short flags;

058    double last_update_time;
059    boolean stop = false;
060    private IMUProtocol.YPRUpdate ypr_update_data;
061    protected byte update_type = IMUProtocol.MSGID_YPR_UPDATE;

063    /**
064     * Constructs the IMU class, overriding the default update rate
065     * with a custom rate which may be from 4 to 100, representing
066     * the number of updates per second sent by the nav6 IMU.  
067     * 
068     * Note that increasing the update rate may increase the 
069     * CPU utilization.
070     * @param serial_port BufferingSerialPort object to use
071     * @param update_rate_hz Custom Update Rate (Hz)
072     */
073    public IMU(BufferingSerialPort serial_port, byte update_rate_hz) {
074        ypr_update_data = new IMUProtocol.YPRUpdate();
075        this.update_rate_hz = update_rate_hz;
076        flags = 0;
077        accel_fsr_g = DEFAULT_ACCEL_FSR_G;
078        gyro_fsr_dps = DEFAULT_GYRO_FSR_DPS;
079        this.serial_port = serial_port;
080        yaw_history = new float[YAW_HISTORY_LENGTH];
081        yaw = (float) 0.0;
082        pitch = (float) 0.0;
083        roll = (float) 0.0;
084        try {
085            serial_port.reset();
086        } catch (VisaException ex) {
087            ex.printStackTrace();
088        }
089        initIMU();
090        m_thread = new Thread(this);
091        m_thread.start();        
092    }
093    
094    /**
095     * Constructs the IMU class, using the default update rate.  
096     * 
097     * @param serial_port BufferingSerialPort object to use
098     */
099    public IMU(BufferingSerialPort serial_port) {
100        this(serial_port,DEFAULT_UPDATE_RATE_HZ);
101    }

103    protected void initIMU() {
104        
105        // The nav6 IMU serial port configuration is 8 data bits, no parity, one stop bit. 
106        // No flow control is used.
107        // Conveniently, these are the defaults used by the WPILib's SerialPort class.
108        //
109        // In addition, the WPILib's SerialPort class also defaults to:
110        //
111        // Timeout period of 5 seconds
112        // Termination ('\n' character)
113        // Transmit immediately
114
115        initializeYawHistory();
116        user_yaw_offset = 0;
117
118        // set the nav6 into the desired update mode
119        byte stream_command_buffer[] = new byte[256];
120        int packet_length = IMUProtocol.encodeStreamCommand( stream_command_buffer, update_type, update_rate_hz ); 
121        try {
122            serial_port.write( stream_command_buffer, packet_length );
123        } catch (VisaException ex) {
124        }
125    }
126
127    protected void setStreamResponse( IMUProtocol.StreamResponse response ) {
128        
129        flags = response.flags;
130        nav6_yaw_offset_degrees = response.yaw_offset_degrees;
131        accel_fsr_g = response.accel_fsr_g;
132        gyro_fsr_dps = response.gyro_fsr_dps;
133        update_rate_hz = (byte)response.update_rate_hz;
134    }
135        
136    private void initializeYawHistory() {
137
138        Arrays.fill(yaw_history,0);
139        next_yaw_history_index = 0;
140        last_update_time = 0.0;
141    }
142
143    private void setYawPitchRoll(float yaw, float pitch, float roll, float compass_heading) {
144
145        this.yaw = yaw;
146        this.pitch = pitch;
147        this.roll = roll;
148        this.compass_heading = compass_heading;
149
150        updateYawHistory(this.yaw);
151    }
152
153    protected void updateYawHistory(float curr_yaw) {
154
155        if (next_yaw_history_index >= YAW_HISTORY_LENGTH) {
156            next_yaw_history_index = 0;
157        }
158        yaw_history[next_yaw_history_index] = curr_yaw;
159        last_update_time = Timer.getFPGATimestamp();
160        next_yaw_history_index++;
161    }
162
163    private double getAverageFromYawHistory() {
164
165        double yaw_history_sum = 0.0;
166        for (int i = 0; i < YAW_HISTORY_LENGTH; i++) {
167            yaw_history_sum += yaw_history[i];
168        }
169        double yaw_history_avg = yaw_history_sum / YAW_HISTORY_LENGTH;
170        return yaw_history_avg;
171    }

191    /**
192     * Returns the current yaw value (in degrees, from -180 to 180)
193     * reported by the nav6 IMU.
194     * 
195     * Note that the returned yaw value will be offset by a user-specified
196     * offset value; this user-specified offset value is set by 
197     * invoking the zeroYaw() method.
198     * @return The current yaw value in degrees (-180 to 180).
199     */
200    public float getYaw() {
201        float calculated_yaw = (float) (this.yaw - user_yaw_offset);
202        if (calculated_yaw < -180) {
203            calculated_yaw += 360;
204        }
205        if (calculated_yaw > 180) {
206            calculated_yaw -= 360;
207        }
208        return calculated_yaw;
209    }
210
211    /**
212     * Returns the current tilt-compensated compass heading 
213     * value (in degrees, from 0 to 360) reported by the nav6 IMU.
214     * 
215     * Note that this value is sensed by the nav6 magnetometer,
216     * which can be affected by nearby magnetic fields (e.g., the
217     * magnetic fields generated by nearby motors).
218     * @return The current tilt-compensated compass heading, in degrees (0-360).
219     */
220    public float getCompassHeading() {
221        return compass_heading;
222    }
223
224    /**
225     * Sets the user-specified yaw offset to the current
226     * yaw value reported by the nav6 IMU.
227     * 
228     * This user-specified yaw offset is automatically
229     * subtracted from subsequent yaw values reported by
230     * the getYaw() method.
231     */
232    public void zeroYaw() {
233        user_yaw_offset = getAverageFromYawHistory();
234    }

236    /**
237     * Indicates whether the nav6 IMU is currently connected
238     * to the host computer.  A connection is considered established
239     * whenever a value update packet has been received from the
240     * nav6 IMU within the last second.
241     * @return Returns true if a valid update has been received within the last second.
242     */
243    public boolean isConnected() {
244        double time_since_last_update = Timer.getFPGATimestamp() - this.last_update_time;
245        return time_since_last_update <= 1.0;
246    }

248    /**
249     * Returns the count in bytes of data received from the
250     * nav6 IMU.  This could can be useful for diagnosing 
251     * connectivity issues.
252     * 
253     * If the byte count is increasing, but the update count
254     * (see getUpdateCount()) is not, this indicates a software
255     * misconfiguration.
256     * @return The number of bytes received from the nav6 IMU.
257     */

259        return byte_count;
260    }

262    /**
263     * Returns the count of valid update packets which have
264     * been received from the nav6 IMU.  This count should increase
265     * at the same rate indicated by the configured update rate.
266     * @return The number of valid updates received from the nav6 IMU.
267     */
268    public double getUpdateCount() {
269        return update_count;
270    }

272    /**
273     * Returns true if the nav6 IMU is currently performing automatic
274     * calibration.  Automatic calibration occurs when the nav6 IMU
275     * is initially powered on, during which time the nav6 IMU should
276     * be held still.
277     * 
278     * During this automatically calibration, the yaw, pitch and roll
279     * values returned may not be accurate.
280     * 
281     * Once complete, the nav6 IMU will automatically remove an internal
282     * yaw offset value from all reported values.
283     * @return Returns true if the nav6 IMU is currently calibrating.
284     */
285    public boolean isCalibrating() {
286        short calibration_state = (short)(this.flags & IMUProtocol.NAV6_FLAG_MASK_CALIBRATION_STATE);
287        return (calibration_state != IMUProtocol.NAV6_CALIBRATION_STATE_COMPLETE);
288    }

325    // Invoked when a new packet is received; returns the packet length if the packet 
326    // is valid, based upon IMU Protocol definitions; otherwise, returns 0
327    
328    protected int decodePacketHandler(byte[] received_data, int offset, int bytes_remaining) {
329        
330        int packet_length = IMUProtocol.decodeYPRUpdate(received_data, offset, bytes_remaining, ypr_update_data);
331        if (packet_length > 0) {
332            setYawPitchRoll(ypr_update_data.yaw,ypr_update_data.pitch,ypr_update_data.roll,ypr_update_data.compass_heading);
333        }
334        return packet_length;
335    }

337    // IMU Class thread run method
338    
339    public void run() {
340
341        stop = false;
342        boolean stream_response_received = false;
343        double last_stream_command_sent_timestamp = 0.0;
344        try {
345            serial_port.setReadBufferSize(512);
346            serial_port.setTimeout(1.0);
347            serial_port.enableTermination('\n');
348            serial_port.flush();
349            serial_port.reset();
350        } catch (VisaException ex) {
351            ex.printStackTrace();
352        }
353                
354        IMUProtocol.StreamResponse response = new IMUProtocol.StreamResponse();
355
356        byte[] stream_command = new byte[256];
357        
358        int cmd_packet_length = IMUProtocol.encodeStreamCommand( stream_command, update_type, update_rate_hz ); 
359        try {
360            serial_port.reset();
361            serial_port.write( stream_command, cmd_packet_length );
362            serial_port.flush();
363            last_stream_command_sent_timestamp = Timer.getFPGATimestamp();
364        } catch (VisaException ex) {
365        }
366        
367        while (!stop) {
368            try {
369                int packets_received = 0;
370                byte[] received_data = serial_port.read(256);
371                int bytes_read = received_data.length;
372                if (bytes_read > 0) {
373                    byte_count += bytes_read;
374                    int i = 0;
375                    // Scan the buffer looking for valid packets
376                    while (i < bytes_read) {
377                                                
378                        // Attempt to decode a packet
379                        
380                        int bytes_remaining = bytes_read - i;
381                        int packet_length = decodePacketHandler(received_data,i,bytes_remaining);
382                        if (packet_length > 0) {
383                            packets_received++;
384                            update_count++;
385                            i += packet_length;
386                        } 
387                        else 
388                        {
389                            packet_length = IMUProtocol.decodeStreamResponse(received_data, i, bytes_remaining, response);
390                            if (packet_length > 0) {
391                                packets_received++;
392                                setStreamResponse(response);
393                                stream_response_received = true;
394                                i += packet_length;
395                            }
396                            else {
397                                // current index is not the start of a valid packet; increment
398                                i++;
399                            }
400                        }
401                    }
402                
403                    if ( ( packets_received == 0 ) && ( bytes_read == 256 ) ) {
404                        // Workaround for issue found in Java SerialPort implementation:
405                        // No packets received and 256 bytes received; this
406                        // condition occurs in the Java SerialPort.  In this case,
407                        // reset the serial port.
408                        serial_port.reset();
409                    }
410                    
411                    // If a stream configuration response has not been received within three seconds
412                    // of operation, (re)send a stream configuration request
413                    
414                    if ( !stream_response_received && ((Timer.getFPGATimestamp() - last_stream_command_sent_timestamp ) > 3.0 ) ) {
415                        cmd_packet_length = IMUProtocol.encodeStreamCommand( stream_command, update_type, update_rate_hz ); 
416                        try {
417                            serial_port.write( stream_command, cmd_packet_length );
418                            serial_port.flush();
419                            last_stream_command_sent_timestamp = Timer.getFPGATimestamp();
420                        } catch (VisaException ex2) {
421                        }                                                    
422                    }
423                    else {
424                        
425                        // If no bytes remain in the buffer, and not awaiting a response, sleep a bit
426                        if ( stream_response_received && ( serial_port.getBytesReceived() == 0 ) ) {
427                            Timer.delay(1.0/update_rate_hz);
428                        }        
429                    }
430                }
431            } catch (VisaException ex) {
432
433                // This exception typically indicates a Timeout
434                
435                try {
436                    double start_wait_timer = Timer.getFPGATimestamp();
437                    int bytes_received = serial_port.getBytesReceived();
438                    
439                    // If no bytes have been received after the timeout period,
440                    // wait for more bytes to arrive
441                    
442                    while ( !stop && ( bytes_received == 0 ) ) {
443                        Timer.delay(1.0/update_rate_hz);
444                        bytes_received = serial_port.getBytesReceived();
445                    }
446                    
447                    if ( !stop && (bytes_received > 0 ) ) {
448                        if ( (Timer.getFPGATimestamp() - start_wait_timer ) > 1.0 ) {
449                            
450                            // Some bytes have arrived, after a timeout
451                            // If > 1 second has gone by since the timeout,
452                            // Assume the nav6 board has been reset; in this case,
453                            // re-issue the stream configuration command
454                            // nav6 initialization takes 2 seconds (worst case)
455                            // so wait a bit before sending the stream configuration command
456                            
457                            stream_response_received = false;
458                            Timer.delay(2.0);
459                            cmd_packet_length = IMUProtocol.encodeStreamCommand( stream_command, update_type, update_rate_hz ); 
460                            try {
461                                serial_port.reset();                                
462                                serial_port.write( stream_command, cmd_packet_length );
463                                serial_port.flush();
464                                last_stream_command_sent_timestamp = Timer.getFPGATimestamp();
465                            } catch (VisaException ex2) {
466                            }                            
467                        }
468                    }
469                } catch (VisaException ex1) {
470                    ex1.printStackTrace();
471                }
472            }
473        }
474    }
